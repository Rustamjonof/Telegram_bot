def tariffs_kb(lang, uid):
    import sys
    main_module = sys.modules.get('__main__')
    sett   = main_module.load_settings() if main_module and hasattr(main_module, 'load_settings') else {}
    custom = sett.get("custom_section_texts", {})

    def gbtn(key, default):
        return custom.get(key, {}).get(lang, default)

    sub  = check_subscription(uid)
    rows = []
    for tariff in ["standard", "premium"]:
        default_name = TARIFF_NAMES.get(lang, TARIFF_NAMES["en"])[tariff]
        name         = gbtn(f"btn_tariff_{tariff}", default_name)
        active       = " ✓" if sub == tariff else ""
        rows.append([InlineKeyboardButton(
            text=f"{name}{active}",
            callback_data=f"tariff_view_{tariff}"
        )])
    rows.append([InlineKeyboardButton(
        text=gbtn("btn_back", BACK_BTN.get(lang, "🔙 Back")),
        callback_data="tariff_back"
    )])
    return InlineKeyboardMarkup(inline_keyboard=rows)

def period_kb(lang, tariff, uid):
    monthly_price, monthly_disc = get_discounted_price(uid, tariff, 'monthly')
    yearly_price,  yearly_disc  = get_discounted_price(uid, tariff, 'yearly')

    monthly_text = f"📅 {PERIOD_NAMES.get(lang, PERIOD_NAMES['en'])['monthly']} — {monthly_price:,} UZS"
    yearly_text  = f"📆 {PERIOD_NAMES.get(lang, PERIOD_NAMES['en'])['yearly']} — {yearly_price:,} UZS"

    if monthly_disc > 0:
        monthly_text += f" (-{monthly_disc}%)"
    if yearly_disc > 0:
        yearly_text += f" (-{yearly_disc}%)"

    return InlineKeyboardMarkup(inline_keyboard=[
        [InlineKeyboardButton(text=monthly_text, callback_data=f"tariff_pay_{tariff}_monthly")],
        [InlineKeyboardButton(text=yearly_text,  callback_data=f"tariff_pay_{tariff}_yearly")],
        [InlineKeyboardButton(text=BACK_BTN.get(lang, "🔙 Back"), callback_data="tariff_list")],
    ])

def payment_method_kb(lang, tariff, period):
    return InlineKeyboardMarkup(inline_keyboard=[
        [InlineKeyboardButton(text=CARD_BTN.get(lang, "💳 Karta"),            callback_data=f"tariff_card_{tariff}_{period}")],
        [InlineKeyboardButton(text=BACK_BTN.get(lang, "🔙 Back"),             callback_data=f"tariff_view_{tariff}")],
    ])

def register_tariff_handlers(dp, bot):

    # ---- TARIFLAR RO'YXATI ----
    @dp.callback_query(F.data == "tariff_list")
    async def cb_tariff_list(callback: types.CallbackQuery):
        uid  = callback.from_user.id
        import sys
        main_module = sys.modules.get('__main__')
        if main_module and hasattr(main_module, 'get_user'):
            user = main_module.get_user(uid)
        else:
            user = get_user(uid)
        lang = user.get("lang", "en")

        sub  = check_subscription(uid)
        tariff_name = TARIFF_NAMES.get(lang, TARIFF_NAMES["en"]).get(sub, sub)
        text = (
            f"{TARIFF_TITLE.get(lang, '💎 Tariffs')}\n\n"
            f"{CURRENT_TARIFF.get(lang, 'Current plan')}: {tariff_name}\n\n"
            f"{TARIFF_FEATURES.get(lang, TARIFF_FEATURES['en']).get(sub, '')}"
        )
        await callback.message.edit_text(text, reply_markup=tariffs_kb(lang, uid))

    @dp.callback_query(F.data == "tariff_back")
    async def cb_tariff_back(callback: types.CallbackQuery):
        await callback.message.delete()

    # ---- TARIF KO'RISH ----

        main_module = sys.modules.get('__main__')
        if main_module and hasattr(main_module, 'get_user'):
            user = main_module.get_user(uid)
        else:
            user = get_user(uid)
        lang = user.get("lang", "en")

        tariff = callback.data[12:]
        name   = TARIFF_NAMES.get(lang, TARIFF_NAMES["en"]).get(tariff, tariff)
        feats  = TARIFF_FEATURES.get(lang, TARIFF_FEATURES["en"]).get(tariff, "")

        yearly_price,  yearly_disc  = get_discounted_price(uid, tariff, 'yearly')

        price_text = (

            f"{yearly_price:,} UZS"
            + (f" (-{yearly_disc}%)" if yearly_disc else "")

        import sys
        main_module  = sys.modules.get('__main__')
        sett         = main_module.load_settings() if main_module and hasattr(main_module, 'load_settings') else {}
        custom_feats = sett.get("custom_section_texts", {}).get(f"tariff_{tariff}", {}).get(lang, "")

        if custom_feats:
            await callback.message.edit_text(custom_feats, reply_markup=period_kb(lang, tariff, uid))
        else:
            await callback.message.edit_text(
                f"{name}\n\n{feats}{price_text}\n\n{CHOOSE_PERIOD.get(lang, 'Choose period:')}",

            f"💳 {tariff_name} — {period_name}\n"
            f"💰 Summa: {price:,} UZS{disc_text}\n\n"

        )

            "ru": "✅ Чек получен!\nПосле подтверждения администратора тариф будет активирован.",
            "ar": "✅ تم استلام الإيصال!\nبعد تأكيد المشرف سيتم تفعيل باقتك.",
            "tr": "✅ Makbuz alındı!\nAdmin onayladıktan sonra planınız aktif olacak.",
            "es": "✅ ¡Recibo recibido!\nDespués de la confirmación del admin, tu plan se activará.",
            "hi": "✅ रसीद प्राप्त हुई!\nएडमिन की पुष्टि के बाद आपकी योजना सक्रिय होगी।",

            await callback.answer("To'lov topilmadi!", show_alert=True)
            return

        await callback.message.edit_reply_markup(
            reply_markup=InlineKeyboardMarkup(inline_keyboard=[[
                InlineKeyboardButton(
                    text="✅ Ha, tasdiqlayman",
                    callback_data=f"tconf2_{payment_id}"
                ),
                InlineKeyboardButton(
                    text="🔙 Orqaga",
                    callback_data=f"tback_{payment_id}"
                ),
            ]])
        )
        await callback.answer("Haqiqatan tasdiqlaysizmi?", show_alert=True)
